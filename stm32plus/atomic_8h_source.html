00003 // ARMV6 , ARMV7 Atomic primitives
00004 // Based on knowledge
00005 // http://infocenter.arm.com/help/index.jsp?topic=/com.arm.doc.dui0489b/Cihbghef.html
00006 //
00007 // Autor Artur Bac 2010 EBASoft
00008 //
00009 // Under BSD License
00010 //
00011 // version 1.0.0
00012 // Written for Bada & GCC C++ but will also work on any other platform with Cortex CPU like IPhone, Android
00013 // ARMv6 introduces a new mechanism, known as "exclusives",
00014 // using the LDREX and STREX instructions.
00015 // Direct use of these instructions is not
00016 // recommended for new code (see "GCC atomic primitives" for a better alternative).
00017 // However, to assist with understanding existing code, a quick overview follows:
00018 //      LDREX   r0, [r1]
00019 //            // do something with r0
00020 //           // no other load or store operations can occur between a LDREX and its corresponding STREX
00021 //       STREX   r2, r0, [r1]
00022 //            // now r2 = 0 if the new value was stored
00023 //            // r2 = 1 if the store was abandoned
00024 //
00025 //Because the STREX is allowed to "fail", it is no longer necessary to lock out other bus activity,
00026 // or stop other threads from executing (whether on the same CPU or on another CPU in a multiprocessor system);
00027 // atomic use in a thread can preempt a pending atomic in another thread.
00028 // For this reason, these primitives are usually also placed in a loop.
00029 //Note that extra load and store instruction (of any kind) between LDREX and STREX can cause the STREX always to fail in some implementations,
00030 //which is why you shouldn't access memory between the LDREX and the corresponding STREX,
00031 //except in cases where it's sensible to abandon the operation such as context switches or exceptions.
00032 
00033 //These builtins perform an atomic compare and swap.
00034 //That is, if the current value of *ptr is oldval, then write newval into *ptr.
00035 //The val version returns the contents of *ptr before the operation.

00050     template<typename T, typename U>
00051     struct sync_bool_compare_and_swap_internal<T, U, sizeof(long)> {
00052       bool operator()(T *ptr, U oldval, U newval) const {
00053         register int result;
00054         asm volatile (
00055             "ldrex    r0, [%1]         \n\t" /*exclusive load of ptr */
00056             "cmp      r0,  %2          \n\t" /*compare the oldval ==  *ptr */
00057             "ite eq\n\t"
00058             "strexeq  %0,  %3, [%1]\n\t" /*store if eq, strex+eq*/
00059             "clrexne"
00060             : "=&r" (result)
00061             : "r"(ptr), "r"(oldval),"r"(newval)
00062             : "r0"
00063         );
00064         return result == 0;
00065       }
00066     };
