00088     void RmiiInterface<TPhysicalLayer,TPinPackage>::outputClockOnMCO() const {
00089 
00090       // set PLL3 clock output to 50MHz (25MHz /5 *10 =50MHz)
00091 
00092       RCC_PLL3Config(RCC_PLL3Mul_10);
00093 
00094       // Enable PLL3
00095 
00096       RCC_PLL3Cmd(ENABLE);
00097 
00098       // Wait till PLL3 is ready
00099 
00100       while (RCC_GetFlagStatus(RCC_FLAG_PLL3RDY)==RESET);
00101 
00102       // Get clock PLL3 clock on PA8 pin
00103 
00104       RCC_MCOConfig(RCC_MCO_PLL3CLK);
00105     }
