00048     /*
00049      * The target for the DMA transfer depends on the alignment, channel (or dual channel).
00050      * All of these are available as compile-time constants in the alignment feature therefore
00051      * the optimiser will eliminate all the conditional tests below, remove the dead code and emit
00052      * only the resulting single storage instructions.
00053      */
00054 
00055     if(TDacAlignmentFeature::DUAL_CHANNEL_FEATURE) {
00056 
00057       // both channels
00058 
00059       if(TDacAlignmentFeature::getAlignment()==DAC_Align_12b_R)
00060         _init.DMA_PeripheralBaseAddr=reinterpret_cast<uint32_t>(&(DAC->DHR12RD));
00061       else if(TDacAlignmentFeature::getAlignment()==DAC_Align_12b_L)
00062         _init.DMA_PeripheralBaseAddr=reinterpret_cast<uint32_t>(&(DAC->DHR12LD));
00063       else if(TDacAlignmentFeature::getAlignment()==DAC_Align_8b_R)
00064         _init.DMA_PeripheralBaseAddr=reinterpret_cast<uint32_t>(&(DAC->DHR8RD));
00065     }
00066     else {
00067       if(TDacAlignmentFeature::getChannel()==DAC_Channel_1) {
00068 
00069         if(TDacAlignmentFeature::getAlignment()==DAC_Align_12b_R)
00070           _init.DMA_PeripheralBaseAddr=reinterpret_cast<uint32_t>(&(DAC->DHR12R1));
00071         else if(TDacAlignmentFeature::getAlignment()==DAC_Align_12b_L)
00072           _init.DMA_PeripheralBaseAddr=reinterpret_cast<uint32_t>(&(DAC->DHR12L1));
00073         else if(TDacAlignmentFeature::getAlignment()==DAC_Align_8b_R)
00074           _init.DMA_PeripheralBaseAddr=reinterpret_cast<uint32_t>(&(DAC->DHR8R1));
00075       }
00076       else if(TDacAlignmentFeature::getChannel()==DAC_Channel_2) {
00077 
00078         if(TDacAlignmentFeature::getAlignment()==DAC_Align_12b_R)
00079           _init.DMA_PeripheralBaseAddr=reinterpret_cast<uint32_t>(&(DAC->DHR12R2));
00080         else if(TDacAlignmentFeature::getAlignment()==DAC_Align_12b_L)
00081           _init.DMA_PeripheralBaseAddr=reinterpret_cast<uint32_t>(&(DAC->DHR12L2));
00082         else if(TDacAlignmentFeature::getAlignment()==DAC_Align_8b_R)
00083           _init.DMA_PeripheralBaseAddr=reinterpret_cast<uint32_t>(&(DAC->DHR8R2));
00084       }
00085     }
00086 
00087     if(TDacAlignmentFeature::DUAL_CHANNEL_FEATURE) {
00088 
00089       // 8 bit = half-word (packed channels: 22222221111111) transfers
00090 
00091       if(TDacAlignmentFeature::getAlignment()==DAC_Align_8b_R) {
00092         _init.DMA_PeripheralDataSize=DMA_PeripheralDataSize_HalfWord;
00093         _init.DMA_MemoryDataSize=DMA_MemoryDataSize_HalfWord;
00094       }
00095       else {
00096 
00097         // 12 bit = word (packed channels: xxxx222222222222xxxx111111111111) transfers
00098 
00099         _init.DMA_PeripheralDataSize=DMA_PeripheralDataSize_Word;
00100         _init.DMA_MemoryDataSize=DMA_MemoryDataSize_Word;
00101       }
00102 
00103     }
00104     else {
00105 
00106       if(TDacAlignmentFeature::getAlignment()==DAC_Align_8b_R) {
00107         _init.DMA_PeripheralDataSize=DMA_PeripheralDataSize_Byte;     // transferring bytes
00108         _init.DMA_MemoryDataSize=DMA_MemoryDataSize_Byte;             // transferring bytes
00109       }
00110       else {
00111         _init.DMA_PeripheralDataSize=DMA_PeripheralDataSize_HalfWord; // transferring half-words
00112         _init.DMA_MemoryDataSize=DMA_MemoryDataSize_HalfWord;         // transferring half-words
00113       }
00114     }

00116     _init.DMA_DIR=DMA_DIR_PeripheralDST;                      // 'peripheral' is destination
