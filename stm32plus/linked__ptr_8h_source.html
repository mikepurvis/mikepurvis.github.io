00042   // This is used internally by all instances of linked_ptr<>.  It needs to be
00043   // a non-template class because different types of linked_ptr<> can refer to
00044   // the same object (linked_ptr<Superclass>(obj) vs linked_ptr<Subclass>(obj)).
00045   // So, it needs to be possible for different types of linked_ptr to participate
00046   // in the same circular linked list, so we need a single class type here.
00047   //
00048   // DO NOT USE THIS CLASS DIRECTLY YOURSELF.  Use linked_ptr<T>.
00049 
00050   class linked_ptr_internal {
00051     public:
00052       // Create a new circle that includes only this instance.
00053       void join_new() {
00054         next_ = this;
00055       }
00056 
00057       // Join an existing circle.
00058       void join(linked_ptr_internal const* ptr) {
00059         next_ = ptr->next_;
00060         ptr->next_ = this;
00061       }
00062 
00063       // Leave whatever circle we're part of.  Returns true iff we were the
00064       // last member of the circle.  Once this is done, you can join() another.
00065       bool depart() {
00066         if (next_ == this)
00067           return true;
00068         linked_ptr_internal const* p = next_;
00069         while (p->next_ != this)
00070           p = p->next_;
00071         p->next_ = next_;
00072         return false;
00073       }
00074 
00075     private:
00076       mutable linked_ptr_internal const* next_;
00077   };
