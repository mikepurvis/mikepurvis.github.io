00049     template<class TPinPackage>
00050     inline Gpio16BitAccessMode<TPinPackage,COLOURS_16BIT,72,50,50>::Gpio16BitAccessMode() {
00051 
00052       // the assembly code needs these
00053 
00054       _rs=TPinPackage::Pin_RS;
00055       _wr=TPinPackage::Pin_WR;
00056 
00057       // these are the addresses of the reset/set registers in the normal peripheral region.
00058 
00059 #if defined(STM32PLUS_F1)
00060       _controlResetAddress=TPinPackage::Port_CONTROL+offsetof(GPIO_TypeDef,BRR);
00061       _controlSetAddress=TPinPackage::Port_CONTROL+offsetof(GPIO_TypeDef,BSRR);
00062 #elif defined(STM32PLUS_F4)
00063       _controlResetAddress=TPinPackage::Port_CONTROL+offsetof(GPIO_TypeDef,BSRRH);
00064       _controlSetAddress=TPinPackage::Port_CONTROL+offsetof(GPIO_TypeDef,BSRRL);
00065 #else
00066 #error Unsupported MCU
00067 #endif
00068 
00069       // this is the address of the data output ODR register in the normal peripheral region.
00070 
00071       _portOutputRegister=TPinPackage::Port_DATA+offsetof(GPIO_TypeDef,ODR);
00072 
00073       // all 16 port pins to output, 50MHz slew rate
00074 
00075       GpioPinInitialiser::initialise((GPIO_TypeDef *)TPinPackage::Port_DATA,
00076                                      0xffff,
00077                                      Gpio::OUTPUT);
00078 
00079       // control pins to output
00080 
00081       GpioPinInitialiser::initialise((GPIO_TypeDef *)TPinPackage::Port_CONTROL,
00082                                      TPinPackage::Pin_RS | TPinPackage::Pin_WR | TPinPackage::Pin_RESET,
00083                                      Gpio::OUTPUT);
00084 
00085       // WR must start as HIGH
00086 
00087       GPIO_SetBits((GPIO_TypeDef *)TPinPackage::Port_CONTROL,TPinPackage::Pin_WR);
00088     }

00122     template<class TPinPackage>
00123     inline void Gpio16BitAccessMode<TPinPackage,COLOURS_16BIT,72,50,50>::writeCommand(uint16_t command) const {
00124 
00125       __asm volatile(
00126         " str  %[value], [%[data]]          \n\t"     // port <= value
00127         " str  %[rs],    [%[creset], #0]    \n\t"     // [rs] = 0
00128         " str  %[wr],    [%[creset], #0]    \n\t"     // [wr] = 0
00129         " mov  r0,       r0                 \n\t"     // burn 2 cycles so we meet the timing requirements
00130         " mov  r0,       r0                 \n\t"
00131         " str  %[wr],    [%[cset], #0]      \n\t"     // [wr] = 1
00132         :: [creset]   "r" (_controlResetAddress),     // the control reset address
00133            [cset]     "r" (_controlSetAddress),       // the control set address
00134            [data]     "r" (_portOutputRegister),      // the data port
00135            [wr]       "r" (_wr),                      // WR pin bit
00136            [rs]       "r" (_rs),                      // RS pin bit
00137            [value]    "r" (command)                   // input value
00138       );
00139     }
