00195 
00196       if(type!=IcmpType::ECHO_REPLY && type!=IcmpType::DESTINATION_UNREACHABLE)

00199       // not interested if the id doesn't match the one we're waiting for
00200       // reply and error messages have the id in the same header position

00202       if(static_cast<IcmpEchoReply&>(rxevent.icmpPacket).icmp_sequenceNumber!=NetUtil::htons(_nextId))
