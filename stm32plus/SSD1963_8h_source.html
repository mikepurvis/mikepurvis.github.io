00070     inline void SSD1963<TOrientation,TColourDepth,TAccessMode,TPanelTraits>::initialise() const {
00071 
00072       const unsigned char FSMC_Bank = 0;
00073 
00074       // slow down the device for the register setup
00075 
00076       FSMC_Bank1->BTCR[FSMC_Bank+1] = FSMC_BTR1_ADDSET_0 | FSMC_BTR1_DATAST_2 | FSMC_BTR1_BUSTURN_0 ;
00077       FSMC_Bank1->BTCR[FSMC_Bank] = FSMC_BCR1_MWID_0 | FSMC_BCR1_WREN | FSMC_BCR1_MBKEN;
00078 
00079       // do a software reset
00080 
00081       _accessMode.writeCommand(ssd1963::SOFTWARE_RESET);
00082       MillisecondTimer::delay(1);
00083 
00084       // reset the device
00085 
00086       _accessMode.reset();
00087 
00088       // set the PLL multipler. the pixel clock is sourced from the PLL.
00089       // PLL frequency = (external oscillator frequency * N) / M
00090       // where 250Mhz < (external oscillator frequency * N) < 800Mhz
00091 
00092       _accessMode.writeCommand(ssd1963::SET_PLL_MN);
00093       _accessMode.writeData(TPanelTraits::PLL_M);
00094       _accessMode.writeData(TPanelTraits::PLL_N);
00095       _accessMode.writeData(4);
00096 
00097       // enable the PLL
00098 
00099       _accessMode.writeCommand(ssd1963::SET_PLL);
00100       _accessMode.writeData(1);
00101       MillisecondTimer::delay(2);
00102       _accessMode.writeCommand(ssd1963::SET_PLL);
00103       _accessMode.writeData(3);
00104       MillisecondTimer::delay(2);
00105 
00106       // do a software reset
00107 
00108       _accessMode.writeCommand(ssd1963::SOFTWARE_RESET);
00109       MillisecondTimer::delay(1);
00110 
00111       // set the pixel clock
00112 
00113       _accessMode.writeCommand(ssd1963::SET_LSHIFT_FREQUENCY);
00114       _accessMode.writeData(TPanelTraits::PIXEL_CLOCK >> 16);
00115       _accessMode.writeData((TPanelTraits::PIXEL_CLOCK >> 8) & 0xff);
00116       _accessMode.writeData(TPanelTraits::PIXEL_CLOCK & 0xff);
00117 
00118       // set up the LCD mode (colour depth, dithering)
00119 
00120       _accessMode.writeCommand(ssd1963::SET_LCD_MODE);
00121       _accessMode.writeData(this->getLcdModeBits());                // 0=18 bit, 1=24 bit
00122       _accessMode.writeData(0);                                     // TFT mode
00123       _accessMode.writeData((TPanelTraits::LONG_SIDE-1) >> 8);      // horizontal size
00124       _accessMode.writeData((TPanelTraits::LONG_SIDE-1) & 0xff);
00125       _accessMode.writeData((TPanelTraits::SHORT_SIDE-1) >> 8);     // vertical size
00126       _accessMode.writeData((TPanelTraits::SHORT_SIDE-1) & 0xff);
00127       _accessMode.writeData(0);                                     // RGB transfer order
00128 
00129       // horizontal timing
00130 
00131       _accessMode.writeCommand(ssd1963::SET_HORIZONTAL_PERIOD);
00132       _accessMode.writeData((TPanelTraits::H_BACK_PORCH+TPanelTraits::H_PERIOD+TPanelTraits::H_FRONT_PORCH) >> 8);
00133       _accessMode.writeData((TPanelTraits::H_BACK_PORCH+TPanelTraits::H_PERIOD+TPanelTraits::H_FRONT_PORCH) & 0xff);
00134       _accessMode.writeData(TPanelTraits::H_BACK_PORCH >> 8);
00135       _accessMode.writeData(TPanelTraits::H_BACK_PORCH & 0xff);
00136       _accessMode.writeData(TPanelTraits::H_SYNC_PULSE_WIDTH-1);
00137       _accessMode.writeData(TPanelTraits::H_SYNC_POSITION >> 8);
00138       _accessMode.writeData(TPanelTraits::H_SYNC_POSITION & 0xff);
00139       _accessMode.writeData(0);
00140 
00141       // vertical timing
00142 
00143       _accessMode.writeCommand(ssd1963::SET_VERTICAL_PERIOD);
00144       _accessMode.writeData((TPanelTraits::V_BACK_PORCH+TPanelTraits::V_PERIOD+TPanelTraits::V_FRONT_PORCH) >> 8);
00145       _accessMode.writeData((TPanelTraits::H_BACK_PORCH+TPanelTraits::V_PERIOD+TPanelTraits::V_FRONT_PORCH) & 0xff);
00146       _accessMode.writeData(TPanelTraits::V_BACK_PORCH >> 8);
00147       _accessMode.writeData(TPanelTraits::V_BACK_PORCH & 0xff);
00148       _accessMode.writeData(TPanelTraits::V_SYNC_PULSE_WIDTH-1);
00149       _accessMode.writeData(TPanelTraits::V_SYNC_POSITION >> 8);
00150       _accessMode.writeData(TPanelTraits::V_SYNC_POSITION & 0xff);
00151       _accessMode.writeData(0);
00152 
00153       // enable packed 12-bit mode (2 16-bit FSMC transfers per pixel). This requires
00154       // that DB0..DB11 are connected. Any other pins exposed by the interface
00155       // (i.e. DB12..23) are unused.
00156 
00157       _accessMode.writeCommand(ssd1963::SET_PIXEL_DATA_INTERFACE);
00158       _accessMode.writeData(1);
00159       MillisecondTimer::delay(5);
00160 
00161       // start up the display
00162 
00163       _accessMode.writeCommand(ssd1963::SLEEP_OUT);
00164       MillisecondTimer::delay(10);
00165       _accessMode.writeCommand(ssd1963::DISPLAY_INVERSION_OFF);
00166       _accessMode.writeCommand(ssd1963::IDLE_MODE_OFF);
00167       _accessMode.writeCommand(ssd1963::NORMAL_DISPLAY_MODE_ON);
00168 
00169       // set the panel orientation
00170 
00171       _accessMode.writeCommand(ssd1963::SET_ADDRESS_MODE,this->getAddressMode());
00172 
00173       // reset the scroll position
00174 
00175       this->setScrollPosition(0);
00176 
00177       // wait the required number of ms. before we can turn the display on
00178 
00179       MillisecondTimer::delay(125);
00180       _accessMode.writeCommand(ssd1963::DISPLAY_ON);
00181 
00182       // the user needs to speed up the FSMC now that this has completed
00183     }
