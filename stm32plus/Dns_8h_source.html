00278                         IpPorts::PORT_DNS_REQUEST,                  // destination port,

00316       uint8_t *answers,*record;
00317 
00318       // the flags must not have an error (unknown host is picked up here as cause=3)
00319 
00320       if((flags & 0xf)!=0)
00321         return this->setError(ErrorProvider::ERROR_PROVIDER_NET_DNS,E_SERVER_ERROR,flags & 0xf);
00322 
00323       // the flags must not indicate truncation
00324 
00325       if((flags & DnsPacketHeader::TRUNCATED)!=0)
00326         return this->setError(ErrorProvider::ERROR_PROVIDER_NET_DNS,E_TRUNCATED_REPLY);
00327 
00328       // there must be at least one answer
00329 
00330       if(NetUtil::ntohs(_replyPacket->dns_numberOfAnswerRrs)==0)
00331         return this->setError(ErrorProvider::ERROR_PROVIDER_NET_DNS,E_NO_ANSWERS);
00332 
00333       // find the answers
00334 
00335       DnsReplyPacket *packet(const_cast<DnsReplyPacket *>(_replyPacket));
00336       answers=packet->findAnswers();
00337 
00338       // find the first A record
00339 
00340       if((record=packet->findRecord(answers,DnsPacketHeader::RecordType::A))==nullptr)
00341         return this->setError(ErrorProvider::ERROR_PROVIDER_NET_DNS,E_NO_A_RECORD_IN_ANSWERS);
00342 
00343       // step over the name (should possibly verify that it's the answer to our question here)
00344 
00345       record=packet->stepOverName(record);
00346 
00347       // pull out the IP address and ttl from the record
00348 
00349       ipAddress.ipAddress=*reinterpret_cast<uint32_t *>(record+10);
00350       ttl=NetUtil::ntohl(*reinterpret_cast<uint32_t *>(record+4));
00351
