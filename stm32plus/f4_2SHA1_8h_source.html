00104 
00105     uint32_t start,count;
00106     const uint32_t *ptr;
00107     HASH_MsgDigest digest;
00108 
00109     // tell the peripheral how much is valid in the last word
00110 
00111     HASH_SetLastWordValidBitsNbr((dataSize % 4)*8);
00112 
00113     // update the digest
00114 
00115     ptr=reinterpret_cast<const uint32_t *>(data);
00116     count=dataSize/4;
00117     if(dataSize % 4)
00118       count++;
00119 
00120     while(count--)
00121       HASH_DataIn(*ptr++);
00122 
00123     // start the digest process
00124 
00125     HASH_StartDigest();
00126 
00127     // wait for BUSY to go low
00128 
00129     if(timeout)
00130       start=MillisecondTimer::millis();
00131 
00132     while(HASH_GetFlagStatus(HASH_FLAG_BUSY)!=RESET) {
00133       if(timeout && MillisecondTimer::hasTimedOut(start,timeout))
00134         return errorProvider.set(ErrorProvider::ERROR_PROVIDER_HASH,E_TIMED_OUT);
00135     }
00136 
00137     // Read the message digest
00138 
00139     HASH_GetDigest(&digest);
00140     memcpy(hash,digest.Data,20);
