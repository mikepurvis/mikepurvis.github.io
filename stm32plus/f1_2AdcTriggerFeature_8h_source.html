00096           reinterpret_cast<TIM_TypeDef *>(PeripheralTraits<TTimerPeripheral>::PERIPHERAL_BASE),
00097             TSource==AdcTriggerSource::Reset ? TIM_TRGOSource_Reset :
00098             TSource==AdcTriggerSource::Enable ? TIM_TRGOSource_Enable :
00099             TSource==AdcTriggerSource::Update ? TIM_TRGOSource_Update :
00100             TSource==AdcTriggerSource::CaptureCompare1 ? TIM_TRGOSource_OC1 :
00101             TSource==AdcTriggerSource::OutputReference1 ? TIM_TRGOSource_OC1Ref :
00102             TSource==AdcTriggerSource::OutputReference2 ? TIM_TRGOSource_OC2Ref :
00103             TSource==AdcTriggerSource::OutputReference3 ? TIM_TRGOSource_OC3Ref :
00104             TIM_TRGOSource_OC4Ref
00105       );
