00067     template<Orientation TOrientation,ColourDepth TColourDepth,class TAccessMode>
00068     inline void SSD1289<TOrientation,TColourDepth,TAccessMode>::initialise() const {
00069 
00070       // reset the device
00071 
00072       _accessMode.reset();
00073 
00074       // init sequence: datashee 15.5 "Display ON Sequence"
00075 
00076       // GON = 1, DTE = 0, D[1:0] = 01
00077       _accessMode.writeCommand(ssd1289::DISPLAY_CONTROL,0x21);
00078 
00079       // oscillator ON
00080       _accessMode.writeCommand(ssd1289::OSCILLATOR,0x01);
00081 
00082       // GON = 1, DTE = 0, D[1:0] = 11
00083       _accessMode.writeCommand(ssd1289::DISPLAY_CONTROL,0x23);
00084 
00085       // Exit sleep mode
00086       _accessMode.writeCommand(ssd1289::SLEEP_MODE,0x00);
00087 
00088       MillisecondTimer::delay(35);
00089 
00090       // GON = 1, DTE = 1, D[1:0] = 11
00091       _accessMode.writeCommand(ssd1289::DISPLAY_CONTROL,0x33);
00092 
00093       // BGR etc.
00094       _accessMode.writeCommand(ssd1289::DRIVER_OUTPUT_CONTROL,0x13f | 0x2000 | 0x800 | 0x200);
00095 
00096       // entry mode: colour and orientation
00097       _accessMode.writeCommand(ssd1289::ENTRY_MODE,this->getColourEntryMode() | this->getOrientationEntryMode());
00098 
00099       // driving waveform control
00100       _accessMode.writeCommand(ssd1289::DRIVING_WAVEFORM_CONTROL,0x600);
00101     }
