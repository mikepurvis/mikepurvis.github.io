00114     inline void ArpCache::internalInsert(const MacAddress& mac,const IpAddress& ip) {
00115 
00116       uint8_t i,found,unused;
00117       uint32_t timeNow;
00118       CacheEntry *ptr;
00119 
00120       // can we release a watcher when this IRQ finishes?
00121 
00122       if(_watchFlag && _watchIp==ip) {
00123         *_foundMac=mac;
00124         _watchFlag=false;
00125       }
00126 
00127       // don't care for the link order here, just walk the array. If a matching
00128       // cache entry is found then get out right away. We also need to take care
00129       // of a change of association so we also try to find matches by IP and MAC
00130       // but not both
00131 
00132       ptr=_array.get();
00133       found=unused=NO_ENTRY;
00134       timeNow=_rtc->getTick();
00135 
00136       for(i=0;i<_insertionPoint;i++) {
00137 
00138         if(ptr->next==NO_ENTRY && ptr->previous==NO_ENTRY && i!=_first)
00139           unused=i;
00140         else {
00141 
00142           // if expired then remove it
00143 
00144           if(timeNow>ptr->expiryTime) {
00145             unlink(i);
00146             unused=i;
00147           }
00148           else if(ptr->ipAddress==ip) {
00149 
00150             if(ptr->macAddress==mac)
00151               found=i;        // found an exact match that hasn't expired
00152             else {
00153 
00154               // IP address is cached but has been assigned to another MAC
00155               // this cache entry must be expunged
00156 
00157               unlink(i);
00158               unused=i;
00159             }
00160           }
00161           else if(ptr->macAddress==mac) {
00162 
00163             // MAC address is in the cache but has a different IP address
00164             // the cache entry is not valid any more
00165 
00166             unlink(i);
00167             unused=i;
00168           }
00169         }
00170         ptr++;
00171       }
00172 
00173       // so what happened? if an unexpired match was found then it needs to come to the front
00174 
00175       if(found!=NO_ENTRY) {
00176 
00177         // take the entry out of the list and put it back at the front
00178 
00179         unlink(found);
00180         insertFront(found);
00181 
00182         return;
00183       }
00184 
00185       // a new entry needs to be created. can we place it in a previously
00186       // used position?
00187 
00188       if(unused!=NO_ENTRY)
00189         i=unused;
00190       else if(_insertionPoint!=_maxEntries) {
00191 
00192         // there is space to put it on the end
00193 
00194         i=_insertionPoint;
00195         _insertionPoint++;
00196       }
00197       else {
00198 
00199         // there is no space, evict the LRU entry at the end of the cache
00200 
00201         i=_last;
00202         unlink(_last);
00203       }
00204 
00205       // enter our details
00206 
00207       ptr=&_array[i];
00208       ptr->macAddress=mac;
00209       ptr->ipAddress=ip;
00210       ptr->expiryTime=_rtc->getTick()+_expirySeconds;
00211 
00212       // we are the most recent, insert at front
00213 
00214       insertFront(i);
00215     }
