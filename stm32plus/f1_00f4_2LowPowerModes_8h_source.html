00085   inline void LowPowerModes::doStop(uint32_t regulatorMode,uint8_t entry) {
00086 
00087     bool systickInterruptEnabled;
00088 
00089     // Systick interrupt cannot be enabled during stop mode
00090     // See: STM32F103xC/D/E errata: "Debugging Stop mode and system tick timer"
00091 
00092     systickInterruptEnabled=(SysTick->CTRL & SysTick_CTRL_TICKINT_Msk) != 0;
00093     if(systickInterruptEnabled)
00094       SysTick->CTRL&=~SysTick_CTRL_TICKINT_Msk;
00095 
00096     // enter stop mode
00097 
00098     PWR_EnterSTOPMode(regulatorMode,entry);
00099 
00100     // re-enable clocks - you must supply this function in your app, and you must have done in order
00101     // to run anything at all though you may have called it something different.
00102 
00103     ::SystemInit();
00104 
00105     // re-enable Systick if it was previously on. If a Systick would have fired during the stop mode
00106     // then that tick is lost, i.e. the interrupt pending bit is not set on wakeup.
00107 
00108     if(systickInterruptEnabled)
00109       SysTick->CTRL|=SysTick_CTRL_TICKINT_Msk;
00110   }
