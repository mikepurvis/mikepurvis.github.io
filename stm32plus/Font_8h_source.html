00069     inline void FontBase::getCharacter(uint8_t character,const FontChar*& fc) const {
00070 
00071       const FontChar *ptr;
00072       int16_t i;
00073 
00074       // the bulk of the characters are in sequential order, see if we can
00075       // index directly into the character array to find it
00076 
00077       ptr=nullptr;
00078 
00079       if(character>=_firstCharacter && character<_lastCharacter) {
00080 
00081         // the character is in range and indexable, is it in sequential place?
00082 
00083         if(character-_firstCharacter<_characterCount && _characters[character-_firstCharacter].Code==character)
00084           ptr=&_characters[character-_firstCharacter];
00085         else {
00086 
00087           // did't find it, search for it going backwards because the likelihood is that
00088           // it's towards the end of the array
00089 
00090           for(i=_characterCount-1;i>=0;i--) {
00091 
00092             if(_characters[i].Code==character) {
00093               ptr=&_characters[i];
00094               break;
00095             }
00096           }
00097         }
00098       }
00099 
00100       if(ptr==nullptr)
00101         ptr=&_characters[0];        // didn't found it, default to first char so user knows something is wrong
00102 
00103       // set up the return data
00104 
00105       fc=ptr;
