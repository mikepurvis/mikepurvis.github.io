00160 
00161     RTC_DateTypeDef rdate;
00162     RTC_TimeTypeDef rtime;
00163     struct tm tstruct;
00164 
00165     // get time and date
00166 
00167     RTC_GetTime(RTC_Format_BIN,&rtime);
00168     RTC_GetDate(RTC_Format_BIN,&rdate);
00169 
00170     // if in 24 hour clock or hour is less than 12 then believe the hours
00171 
00172     if(_hourFormat==RTC_HourFormat_24 || rtime.RTC_H12==RTC_H12_AM)
00173       tstruct.tm_hour=rtime.RTC_Hours;
00174     else
00175       tstruct.tm_hour=12+rtime.RTC_Hours;     // adjust hours + 12 for PM
00176 
00177     tstruct.tm_min=rtime.RTC_Minutes;
00178     tstruct.tm_sec=rtime.RTC_Seconds;
00179 
00180     tstruct.tm_year=rdate.RTC_Year+100;       // years since 1900. 00+100=2000
00181     tstruct.tm_mon=rdate.RTC_Month-1;
00182     tstruct.tm_mday=rdate.RTC_Date;
00183 
00184     return mktime(&tstruct)-946684800;        // subtract our base
00185   }
