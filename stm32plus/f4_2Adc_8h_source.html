00067   inline Adc::Adc(ADC_TypeDef *peripheralAddress,Adc *master)
00068     : _peripheralAddress(peripheralAddress),

00070 
00071     // initialise the ranks back to 1 so that the channel features are ready
00072 
00073     _regularChannelRank[0]=_regularChannelRank[1]=_regularChannelRank[2]=1;
00074     _injectedChannelRank[0]=_injectedChannelRank[1]=_injectedChannelRank[2]=1;

00085     _init.ADC_DataAlign=ADC_DataAlign_Right;
00086     _init.ADC_NbrOfConversion=0;

00088     // set up the default common init values
