00074       void innerHash(uint32_t* result,uint32_t* w) {
00075 
00076         uint32_t a=result[0];
00077         uint32_t b=result[1];
00078         uint32_t c=result[2];
00079         uint32_t d=result[3];
00080         uint32_t e=result[4];
00081 
00082         int round=0;
00083 
00084   #define sha1macro(func,val) \
00085         { \
00086           const uint32_t t = rol(a, 5) + (func) + e + val + w[round]; \
00087           e = d; \
00088           d = c; \
00089           c = rol(b, 30); \
00090           b = a; \
00091           a = t; \
00092         }
00093 
00094         while(round < 16) {
00095           sha1macro((b & c) | (~b & d),0x5a827999)
00096           ++round;
00097         }
00098         while(round < 20) {
00099           w[round]=rol((w[round - 3] ^ w[round - 8] ^ w[round - 14] ^ w[round - 16]),1);
00100           sha1macro((b & c) | (~b & d),0x5a827999)
00101           ++round;
00102         }
00103         while(round < 40) {
00104           w[round]=rol((w[round - 3] ^ w[round - 8] ^ w[round - 14] ^ w[round - 16]),1);
00105           sha1macro(b ^ c ^ d,0x6ed9eba1)
00106           ++round;
00107         }
00108         while(round < 60) {
00109           w[round]=rol((w[round - 3] ^ w[round - 8] ^ w[round - 14] ^ w[round - 16]),1);
00110           sha1macro((b & c) | (b & d) | (c & d),0x8f1bbcdc)
00111           ++round;
00112         }

00130       void hash(const void* src,const int bytelength,uint8_t* hash) {
00131 
00132         // Init the result array.
00133         uint32_t result[5]= { 0x67452301,0xefcdab89,0x98badcfe,0x10325476,0xc3d2e1f0 };
00134 
00135         // Cast the void src pointer to be the byte array we can work with.
00136         const uint8_t* sarray=(const uint8_t*)src;
00137 
00138         // The reusable round buffer
00139         uint32_t w[80];
00140 
00141         // Loop through all complete 64byte blocks.
00142         const int endOfFullBlocks=bytelength - 64;
00143         int endCurrentBlock;
00144         int currentBlock=0;
00145 
00146         while(currentBlock <= endOfFullBlocks) {
00147           endCurrentBlock=currentBlock + 64;
00148 
00149           // Init the round buffer with the 64 byte block data.
00150           for(int roundPos=0;currentBlock < endCurrentBlock;currentBlock+=4) {
00151             // This line will swap endian on big endian and keep endian on little endian.
00152             w[roundPos++]=(uint32_t)sarray[currentBlock + 3] | (((uint32_t)sarray[currentBlock + 2]) << 8) | (((uint32_t)sarray[currentBlock + 1]) << 16) | (((uint32_t)sarray[currentBlock]) << 24);
00153           }
00154           innerHash(result,w);
00155         }
00156 
00157         // Handle the last and not full 64 byte block if existing.
00158         endCurrentBlock=bytelength - currentBlock;
00159         clearWBuffert(w);
00160         int lastBlockBytes=0;
00161         for(;lastBlockBytes < endCurrentBlock;++lastBlockBytes) {
00162           w[lastBlockBytes >> 2]|=(uint32_t)sarray[lastBlockBytes + currentBlock] << ((3 - (lastBlockBytes & 3)) << 3);
00163         }
00164         w[lastBlockBytes >> 2]|=0x80 << ((3 - (lastBlockBytes & 3)) << 3);
00165         if(endCurrentBlock >= 56) {
00166           innerHash(result,w);
00167           clearWBuffert(w);
00168         }
00169         w[15]=bytelength << 3;
00170         innerHash(result,w);
00171 
00172         // Store hash in result pointer, and make sure we get in in the correct order on both endian models.
00173         for(int hashByte=20;--hashByte >= 0;) {
00174           hash[hashByte]=(result[hashByte >> 2] >> (((3 - hashByte) & 0x3) << 3)) & 0xff;
00175         }
00176       }
00177   };
