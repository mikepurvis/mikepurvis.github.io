00113   template<PeripheralName TTimerPeripheral,AdcTriggerSource TSource,uint32_t TEvent,uint32_t TEdge>
00114   struct AdcTimerTriggerFeature : AdcTriggerFeature<TEvent,TEdge>  {
00115 
00121     AdcTimerTriggerFeature(Adc& adc)
00122       : AdcTriggerFeature<TEvent,TEdge>(adc) {
00123 
00124       static_assert(TSource>=AdcTriggerSource::Reset && TSource<=AdcTriggerSource::OutputReference4,"Invalid trigger source");
00125 
00126       TIM_SelectOutputTrigger(
00127           reinterpret_cast<TIM_TypeDef *>(PeripheralTraits<TTimerPeripheral>::PERIPHERAL_BASE),
00128           TSource==AdcTriggerSource::Reset ? TIM_TRGOSource_Reset :
00129           TSource==AdcTriggerSource::Enable ? TIM_TRGOSource_Enable :
00130           TSource==AdcTriggerSource::Update ? TIM_TRGOSource_Update :
00131           TSource==AdcTriggerSource::CaptureCompare1 ? TIM_TRGOSource_OC1 :
00132           TSource==AdcTriggerSource::OutputReference1 ? TIM_TRGOSource_OC1Ref :
00133           TSource==AdcTriggerSource::OutputReference2 ? TIM_TRGOSource_OC2Ref :
00134           TSource==AdcTriggerSource::OutputReference3 ? TIM_TRGOSource_OC3Ref :
00135           TIM_TRGOSource_OC4Ref
00136       );
00137     }
00138   };
