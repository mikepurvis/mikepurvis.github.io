00126     inline bool IpPorts::ip_acquireEphemeralPort(uint16_t& portNumber) {
00127 
00128       uint16_t end;
00129 
00130       // got to stop if we come back here
00131 
00132       end=_nextEphemeralPort;
00133 
00134       // try to find a port. this is not an efficient search method if the number of ephemeral
00135       // ports is high. speed is traded for space.
00136 
00137       while(ip_isEphemeralPortInUse(_nextEphemeralPort)) {
00138 
00139         incrementNextEphemeralPort();
00140 
00141         if(end==_nextEphemeralPort)
00142           return errorProvider.set(ErrorProvider::ERROR_PROVIDER_NET_IPPORTS,E_ALL_PORTS_IN_USE);
00143       }
00144 
00145       // found one
00146 
00147       portNumber=_nextEphemeralPort;
00148       _ephemeralPortsInUse.push_back(portNumber);
00149       incrementNextEphemeralPort();
00150 
00151       return true;
