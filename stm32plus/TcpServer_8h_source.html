00133     __attribute__((noinline)) inline void TcpServer<TConnection,TUser>::onReceive(TcpSegmentEvent& event) {
00134 
00135       // we are only interested in segments sent to our port
00136 
00137       if(event.destinationPort!=_listeningPort)
00138         return;
00139 
00140       // we are only interested in segments that have the SYN flag set (but no ACK)
00141 
00142       if(!(event.tcpHeader.hasSyn() && !event.tcpHeader.hasAck()))
00143         return;
00144 
00145       // this event is handled here
00146 
00147       event.handled=true;
00148 
00149       // we are doing nothing unless we are started
00150 
00151       if(!_started)
00152         return;
00153 
00154       // if an existing connection already has this source/dest port combo then this
00155       // segment is a retransmit and we're going to drop it
00156 
00157       TcpFindConnectionNotificationEvent findconn(event.ipPacket.header->ip_sourceAddress,event.sourcePort,event.destinationPort);
00158       _networkUtilityObjects.NetworkNotificationEventSender.raiseEvent(findconn);
00159 
00160       // if there's a connection, return now as we have a dupe SYN
00161 
00162       if(findconn.tcpConnection!=nullptr)
00163         return;
00164 
00165       // if we've hit the maximum number of connections then we have to ignore it
00166 
00167       if(_connectionCount==_params.tcp_maxConnectionsPerServer)
00168         return;
00169 
00170       // new connection to record
00171 
00172       _connectionCount++;
00173 
00174       // create a new TcpConnection to handle the interaction
00175       // there is 2 phase construction here so that the user's derivation of TcpConnection does
00176       // not have to be aware of the objects required by the base class
00177 
00178       TConnection *connection=ConnectionCreator<TConnection,TUser>::createConnection(_connectionParams,_userptr);
00179 
00180       if(!connection->initialise(
00181             _networkUtilityObjects,
00182             _tcpEvents,
00183             event,
00184             _segmentSizeLimit,
00185             _additionalHeaderSize)) {
00186 
00187         delete connection;
00188         return;
00189       }
00190 
00191       // send a notification event so the caller can claim the connection
00192 
00193       TcpAcceptEvent acceptEvent(*this,connection);
00194       TcpAcceptEventSender.raiseEvent(acceptEvent);
00195 
00196       // if the caller does not want this connection then we delete it
00197 
00198       if(!acceptEvent.accepted)
00199         delete connection;
