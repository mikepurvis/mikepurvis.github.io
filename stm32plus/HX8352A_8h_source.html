00121       _accessMode.writeCommand(hx8352a::POWER_CONTROL_6,0x14);      // VCOMG=0, VDV=1_0100
00122       MillisecondTimer::delay(10);
00123 
00124       _accessMode.writeCommand(hx8352a::POWER_CONTROL_1,0x02);    // GASENB=0, PON=0, DK=0, XDK=0, VLCD_TRI=1, STB=0
00125       MillisecondTimer::delay(10);
00126 
00127       _accessMode.writeCommand(hx8352a::POWER_CONTROL_1,0x0A);    // GASENB=0, PON=0, DK=1, XDK=0, VLCD_TRI=1, STB=0
00128       MillisecondTimer::delay(10);
00129 
00130       _accessMode.writeCommand(hx8352a::POWER_CONTROL_3,0x40);    // AP=000
00131       MillisecondTimer::delay(10);
00132 
00133       _accessMode.writeCommand(hx8352a::SOURCE_CONTROL_1,0x00);   // N_SAP=1100 0000
00134       MillisecondTimer::delay(10);
00135 
00136       // into STB mode
