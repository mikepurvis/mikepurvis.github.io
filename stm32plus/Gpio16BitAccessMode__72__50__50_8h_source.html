00061       // these are the addresses of the reset/set registers in the normal peripheral region.

00064       _controlResetAddress=TPinPackage::Port_CONTROL+offsetof(GPIO_TypeDef,BRR);

00072 
00073       // this is the address of the data output ODR register in the normal peripheral region.
00074 
00075       _portOutputRegister=TPinPackage::Port_DATA+offsetof(GPIO_TypeDef,ODR);
00076 
00077       // all 16 port pins to output, 50MHz slew rate
00078 
00079       GpioPinInitialiser::initialise((GPIO_TypeDef *)TPinPackage::Port_DATA,
00080                                      0xffff,
00081                                      Gpio::OUTPUT);

00083       // control pins to output

00086                                      TPinPackage::Pin_RS | TPinPackage::Pin_WR | TPinPackage::Pin_RESET,

00128 
00129       __asm volatile(
00130         " str  %[value], [%[data]]          \n\t"     // port <= value
00131         " str  %[rs],    [%[creset], #0]    \n\t"     // [rs] = 0
00132         " str  %[wr],    [%[creset], #0]    \n\t"     // [wr] = 0
00133         " mov  r0,       r0                 \n\t"     // burn 2 cycles so we meet the timing requirements
00134         " mov  r0,       r0                 \n\t"
00135         " str  %[wr],    [%[cset], #0]      \n\t"     // [wr] = 1
00136         :: [creset]   "r" (_controlResetAddress),     // the control reset address
00137            [cset]     "r" (_controlSetAddress),       // the control set address
00138            [data]     "r" (_portOutputRegister),      // the data port
