00029   template<class C>
00030   class scoped_ptr {
00031     public:
00032 
00033       // The element type
00034       typedef C element_type;
00035 
00036       // Constructor.  Defaults to intializing with nullptr.
00037       // There is no way to create an uninitialized scoped_ptr.
00038       // The input parameter must be allocated with new.
00039       explicit scoped_ptr(C* p=nullptr) :
00040           ptr_(p) {
00041       }
00042 
00043       // Destructor.  If there is a C object, delete it.
00044       // We don't need to test ptr_ == nullptr because C++ does that for us.
00045       ~scoped_ptr() {
00046         enum {
00047           type_must_be_complete=sizeof(C)
00048         };
00049         delete ptr_;
00050       }
00051 
00052       // Reset.  Deletes the current owned object, if any.
00053       // Then takes ownership of a new object, if given.
00054       // this->reset(this->get()) works.
00055       void reset(C* p=nullptr) {
00056         if(p != ptr_) {
00057           enum {
00058             type_must_be_complete=sizeof(C)
00059           };
00060           delete ptr_;
00061           ptr_=p;
00062         }
00063       }
00064 
00065       // Accessors to get the owned object.
00066       C& operator*() const {
00067         return *ptr_;
00068       }
00069       C* operator->() const {
00070         return ptr_;
00071       }
00072       C* get() const {
00073         return ptr_;
00074       }
00075 
00076       C*& address() {
00077         return ptr_;
00078       }
00079 
00080       // Comparison operators.
00081       // These return whether two scoped_ptr refer to the same object, not just to
00082       // two different but equal objects.
00083       bool operator==(C* p) const {
00084         return ptr_ == p;
00085       }
00086       bool operator!=(C* p) const {
00087         return ptr_ != p;
00088       }
00089 
00090       // Swap two scoped pointers.
00091       void swap(scoped_ptr& p2) {
00092         C* tmp=ptr_;
00093         ptr_=p2.ptr_;
00094         p2.ptr_=tmp;
00095       }
00096 
00097       // Release a pointer.
00098       // The return value is the current pointer held by this object.
00099       // If this object holds a nullptr pointer, the return value is nullptr.
00100       // After this operation, this object will hold a nullptr pointer,
00101       // and will not own the object any more.
00102       C* release() {
00103         C* retVal=ptr_;
00104         ptr_=nullptr;
00105         return retVal;
00106       }
00107 
00108     private:
00109       C* ptr_;
00110 
00111       // Forbid comparison of scoped_ptr types.  If C2 != C, it totally doesn't
00112       // make sense, and if C2 == C, it still doesn't make sense because you should
00113       // never have the same object owned by two different scoped_ptrs.
00114       template<class C2> bool operator==(scoped_ptr<C2> const& p2) const;
00115       template<class C2> bool operator!=(scoped_ptr<C2> const& p2) const;
00116 
00117       // Disallow evil constructors
00118       scoped_ptr(const scoped_ptr&);

00120   };
