00068   inline Adc::Adc(ADC_TypeDef *peripheralAddress,Adc *master)
00069     : _peripheralAddress(peripheralAddress) {
00070 
00071     // store the master ADC if this is dual mode or nullptr if in single
00072 
00073     _master=master;
00074 
00075     // not calibrated yet
00076 
00077     _calibrated=false;
00078 
00079     // initialise the ranks back to 1 so that the channel features are ready
00080 
00081     _regularChannelRank[0]=_regularChannelRank[1]=_regularChannelRank[2]=1;
00082     _injectedChannelRank[0]=_injectedChannelRank[1]=_injectedChannelRank[2]=1;

00086     // the features can customise this before the AdcPeripheral class uses
00087

00091     _init.ADC_ExternalTrigConv=ADC_ExternalTrigConv_None;  // software trigger (features override this)

00100

00103     ADC_ResetCalibration(_peripheralAddress);
