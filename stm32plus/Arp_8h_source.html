00186     inline bool Arp<TDatalinkLayer>::handleIncomingFrame(const DatalinkFrame& frame) {
00187 
00188       ArpFrameData *afd,*reply;
00189       NetBuffer *nb;
00190 
00191       afd=reinterpret_cast<ArpFrameData *>(frame.payload);
00192 
00193       // we respond to ARP requests (code 1)
00194 
00195       if(afd->arp_operation==ArpOperation::REQUEST) {
00196 
00197         // are they asking for our MAC address?
00198 
00199         if(_myIpAddress!=afd->arp_targetProtocolAddress)
00200           return true;      // not an error
00201 
00202         // we have to drop the request if nobody has told us what our address and IP are
00203 
00204         if(!_myMacAddress.isValid() || !_myIpAddress.isValid())
00205           return this->setError(ErrorProvider::ERROR_PROVIDER_NET_ARP,E_UNCONFIGURED);
00206 
00207         // create a net buffer for the response
00208         // ARP responses are generated automatically in response to an IRQ
00209 
00210         nb=new NetBuffer(
00211             this->getDatalinkTransmitHeaderSize(),
00212             sizeof(ArpFrameData),
00213             nullptr);
00214 
00215         // the ARP frame data is the last in the buffer, and therefore the first
00216         // to be written to
00217 
00218         reply=reinterpret_cast<ArpFrameData *>(nb->moveWritePointerBack(sizeof(ArpFrameData)));
00219         reply->initialise();
00220         reply->createReply(*afd,_myMacAddress,_myIpAddress);
00221 
00222         // raise a transmit event for the datalink layer to pick up
00223 
00224         this->NetworkSendEventSender.raiseEvent(
00225               EthernetTransmitRequestEvent(nb,reply->arp_targetHardwareAddress,EtherType::ARP,DatalinkChecksum::IP_HEADER_AND_PROTOCOL));
00226       }
00227       else if(afd->arp_operation==ArpOperation::REPLY) {
00228 
00229         // it's a reply, verify that someone is not claiming to own our address
00230 
00231         if(afd->arp_senderProtocolAddress==_myIpAddress)
00232           return this->setError(ErrorProvider::ERROR_PROVIDER_NET_ARP,E_IP_ADDRESS_CLASH);
00233         else
00234           _arpCache.insert(afd->arp_senderHardwareAddress,afd->arp_senderProtocolAddress);
00235       }
00236 
00237       // send an event to anyone that's interested
00238 
00239       ArpReceiveEventSender.raiseEvent(ArpReceiveEvent(*afd));
00240       return true;
00241     }

00317     template<class TDatalinkLayer>
00318     inline bool Arp<TDatalinkLayer>::handleAddressMappingRequest(ArpMappingRequestEvent& event) {
00319 
00320       IpAddress ip;
00321       uint8_t retry;
00322 
00323       // can handle requests for 255.255.255.255 with no configuration information
00324 
00325       if(event.ipAddress.isBroadcast()) {
00326         event.macAddress=MacAddress::createBroadcastAddress();
00327         event.found=true;
00328         return true;
00329       }
00330 
00331       // we can handle requests for a multicast MAC with no configuration information
00332       // this gives us level 1 (tx only) conformance with RFC1112
00333 
00334       if(event.ipAddress.isMulticastGroup()) {
00335         event.macAddress.createMulticastAddress(event.ipAddress.ipAddressBytes);
00336         event.found=true;
00337         return true;
00338       }
00339 
00340       // must be configured with a subnet mask and gateway
00341 
00342       if(!_mySubnetMask.isValid() || !_defaultGatewayAddress.isValid())
00343         return this->setError(ErrorProvider::ERROR_PROVIDER_NET_ARP,E_UNCONFIGURED);
00344 
00345       // use the event IP address or the gateway as the lookup
00346 
00347       if(_mySubnetMask.matches(event.ipAddress,_defaultGatewayAddress)) {
00348 
00349         // is the IP address the broadcast address for this subnet?
00350 
00351         if(_mySubnetMask.isBroadcastAddress(event.ipAddress)) {
00352           event.macAddress=MacAddress::createBroadcastAddress();
00353           event.found=true;
00354           return true;
00355         }
00356         else
00357           ip=event.ipAddress;           // a local IP address
00358       }
00359       else
00360         ip=_defaultGatewayAddress;      // a remote IP address
00361 
00362       // first check the cache and return if found or we cannot do a lookup
00363 
00364       if((event.found=_arpCache.findMacAddress(ip,event.macAddress)))
00365         return true;
00366 
00367       // if we're in an IRQ then we can't continue because we'd need to do a tx/rx
00368 
00369       if(Nvic::isAnyIrqActive())
00370         return this->setError(ErrorProvider::ERROR_PROVIDER_NET_ARP,E_REQUEST_NOT_PERMITTED);
00371 
00372       for(retry=0;retry<_params.arp_retries;retry++) {
00373 
00374         _arpCache.setWatchIp(ip,&event.macAddress);
00375 
00376         // send off the query
00377 
00378         arpSendRequest(ip);
00379 
00380         // wait for our watcher to be triggered, or a timeout
00381 
00382         if((event.found=_arpCache.waitForWatch(_params.arp_replyTimeout)))
00383           return true;
00384       }
00385 
00386       return this->setError(ErrorProvider::ERROR_PROVIDER_NET_ARP,E_TIMED_OUT);
00387     }
