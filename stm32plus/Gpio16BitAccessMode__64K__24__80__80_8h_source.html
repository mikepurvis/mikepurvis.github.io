00051     template<class TPinPackage>
00052     inline Gpio16BitAccessMode<TPinPackage,COLOURS_16BIT,24,80,80>::Gpio16BitAccessMode() {
00053 
00054       // the assembly code needs these
00055 
00056       _rs=TPinPackage::Pin_RS;
00057       _wr=TPinPackage::Pin_WR;
00058 
00059       // these are the addresses of the reset/set registers in the normal peripheral region.
00060 
00061 #if defined(STM32PLUS_F1) || defined(STM32PLUS_F0)
00062       _controlResetAddress=TPinPackage::Port_CONTROL+offsetof(GPIO_TypeDef,BRR);
00063       _controlSetAddress=TPinPackage::Port_CONTROL+offsetof(GPIO_TypeDef,BSRR);
00064 #elif defined(STM32PLUS_F4)
00065       _controlResetAddress=TPinPackage::Port_CONTROL+offsetof(GPIO_TypeDef,BSRRH);
00066       _controlSetAddress=TPinPackage::Port_CONTROL+offsetof(GPIO_TypeDef,BSRRL);
00067 #else
00068 #error Unsupported MCU
00069 #endif
00070 
00071       // this is the address of the data output ODR register in the normal peripheral region.
00072 
00073       _portOutputRegister=TPinPackage::Port_DATA+offsetof(GPIO_TypeDef,ODR);
00074 
00075       // all 16 port pins to output, 50MHz slew rate
00076 
00077       GpioPinInitialiser::initialise((GPIO_TypeDef *)TPinPackage::Port_DATA,
00078                                      0xffff,
00079                                      Gpio::OUTPUT);
00080 
00081       // control pins to output
00082 
00083       GpioPinInitialiser::initialise((GPIO_TypeDef *)TPinPackage::Port_CONTROL,
00084                                      TPinPackage::Pin_RS | TPinPackage::Pin_WR | TPinPackage::Pin_RESET,
00085                                      Gpio::OUTPUT);
00086 
00087       // WR must start as HIGH
00088 
00089       GPIO_SetBits((GPIO_TypeDef *)TPinPackage::Port_CONTROL,TPinPackage::Pin_WR);
00090     }

00125     inline void Gpio16BitAccessMode<TPinPackage,COLOURS_16BIT,24,80,80>::writeCommand(uint16_t command) const {
00126 
00127       __asm volatile(
00128         " str  %[value], [%[data]]          \n\t"     // port <= value
00129         " str  %[rs],    [%[creset], #0]    \n\t"     // [rs] = 0
00130         " str  %[wr],    [%[creset], #0]    \n\t"     // [wr] = 0
00131         " str  %[wr],    [%[cset], #0]      \n\t"     // [wr] = 1
00132         :: [creset]   "r" (_controlResetAddress),     // the control reset address
00133            [cset]     "r" (_controlSetAddress),       // the control set address
00134            [data]     "r" (_portOutputRegister),      // the data port
00135            [wr]       "r" (_wr),                      // WR pin bit
00136            [rs]       "r" (_rs),                      // RS pin bit
00137            [value]    "r" (command)                   // input value
00138       );
00139     }
